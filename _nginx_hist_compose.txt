name: ministry-platform-prod

services:
  postgres:
    image: postgres:16-alpine
    container_name: ministry_postgres_prod
    restart: unless-stopped
    environment:
      POSTGRES_USER: ${POSTGRES_USER}
      POSTGRES_PASSWORD: ${POSTGRES_PASSWORD}
      POSTGRES_DB: ${POSTGRES_DB}
    volumes:
      - postgres_prod_data:/var/lib/postgresql/data
      - ./infra/postgres/init:/docker-entrypoint-initdb.d:ro
      - ./infra/postgres/backups:/backups
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${POSTGRES_USER} -d ${POSTGRES_DB}"]
      interval: 15s
      timeout: 5s
      retries: 10
    networks:
      - ministry_net_prod

  redis:
    image: redis:7-alpine
    container_name: ministry_redis_prod
    restart: unless-stopped
    command: ["redis-server", "--appendonly", "yes", "--appendfsync", "everysec", "--save", "60", "1000"]
    volumes:
      - redis_prod_data:/data
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 15s
      timeout: 5s
      retries: 10
    networks:
      - ministry_net_prod

  migrate:
    build:
      context: ./services/api
      dockerfile: Dockerfile
      target: migrate
    container_name: ministry_migrate_prod
    env_file:
      - .env.production
    environment:
      DATABASE_URL: postgresql://${POSTGRES_USER}:${POSTGRES_PASSWORD}@postgres:5432/${POSTGRES_DB}?schema=public
    depends_on:
      postgres:
        condition: service_healthy
    restart: "no"
    networks:
      - ministry_net_prod

  api:
    build:
      context: ./services/api
      dockerfile: Dockerfile
    image: ministry-api:${IMAGE_TAG:-latest}
    container_name: ministry_api_prod
    restart: unless-stopped
    init: true
    env_file:
      - .env.production
    environment:
      NODE_ENV: production
      PORT: 4000
      DATABASE_URL: postgresql://${POSTGRES_USER}:${POSTGRES_PASSWORD}@postgres:5432/${POSTGRES_DB}?schema=public
      REDIS_URL: redis://redis:6379
      REDIS_ADAPTER_ENABLED: "true"
      WEBSOCKET_ONLY_MODE: "false"
    volumes:
      - uploads_prod_data:/app/uploads
      - /etc/wopp/firebase-admin.json:/etc/wopp/firebase-admin.json:ro
    depends_on:
      migrate:
        condition: service_completed_successfully
      redis:
        condition: service_healthy
    healthcheck:
      test:
        [
          "CMD",
          "node",
          "-e",
          "require('http').get('http://127.0.0.1:4000/api/v1/health', (r) => process.exit(r.statusCode===200?0:1)).on('error',()=>process.exit(1))",
        ]
      interval: 15s
      timeout: 5s
      retries: 10
      start_period: 30s
    networks:
      - ministry_net_prod

  websocket:
    build:
      context: ./services/api
      dockerfile: Dockerfile
    image: ministry-api:${IMAGE_TAG:-latest}
    container_name: ministry_ws_prod
    restart: unless-stopped
    init: true
    env_file:
      - .env.production
    environment:
      NODE_ENV: production
      PORT: 4100
      DATABASE_URL: postgresql://${POSTGRES_USER}:${POSTGRES_PASSWORD}@postgres:5432/${POSTGRES_DB}?schema=public
      REDIS_URL: redis://redis:6379
      REDIS_ADAPTER_ENABLED: "true"
      WEBSOCKET_ONLY_MODE: "true"
    volumes:
      - uploads_prod_data:/app/uploads
      - /etc/wopp/firebase-admin.json:/etc/wopp/firebase-admin.json:ro
    depends_on:
      migrate:
        condition: service_completed_successfully
      redis:
        condition: service_healthy
    healthcheck:
      test:
        [
          "CMD",
          "node",
          "-e",
          "require('http').get('http://127.0.0.1:4100/api/v1/health', (r) => process.exit(r.statusCode===200?0:1)).on('error',()=>process.exit(1))",
        ]
      interval: 15s
      timeout: 5s
      retries: 10
      start_period: 30s
    networks:
      - ministry_net_prod

  admin-web:
    build:
      context: ./apps/admin-web
      dockerfile: Dockerfile
      args:
        NEXT_PUBLIC_API_BASE_URL: ${NEXT_PUBLIC_API_BASE_URL}
        NEXT_PUBLIC_WEBSOCKET_URL: ${NEXT_PUBLIC_WEBSOCKET_URL}
    image: ministry-admin:${IMAGE_TAG:-latest}
    container_name: ministry_admin_web_prod
    restart: unless-stopped
    init: true
    env_file:
      - .env.production
    environment:
      NODE_ENV: production
      PORT: 3001
      NEXT_PUBLIC_API_BASE_URL: ${NEXT_PUBLIC_API_BASE_URL}
      NEXT_PUBLIC_WEBSOCKET_URL: ${NEXT_PUBLIC_WEBSOCKET_URL}
    depends_on:
      api:
        condition: service_healthy
      websocket:
        condition: service_healthy
    healthcheck:
      test:
        [
          "CMD",
          "node",
          "-e",
          "require('http').get('http://127.0.0.1:3001', (r) => process.exit(r.statusCode===200?0:1)).on('error',()=>process.exit(1))",
        ]
      interval: 15s
      timeout: 5s
      retries: 10
      start_period: 45s
    networks:
      - ministry_net_prod

  reverse-proxy:
    image: nginx:1.27-alpine
    container_name: ministry_nginx_prod
    restart: unless-stopped
    ports:
      - "80:80"
      - "443:443"
      - "8080:8080"
    volumes:
      - ./infra/nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./infra/nginx/conf.d:/etc/nginx/conf.d:ro
      - ./infra/nginx/certs:/etc/nginx/certs:ro
    depends_on:
      api:
        condition: service_healthy
      websocket:
        condition: service_healthy
      admin-web:
        condition: service_healthy
    healthcheck:
      test: ["CMD-SHELL", "wget -qO- http://127.0.0.1:8080/health/nginx || exit 1"]
      interval: 15s
      timeout: 5s
      retries: 10
    networks:
      - ministry_net_prod

volumes:
  postgres_prod_data:
  redis_prod_data:
  uploads_prod_data:

networks:
  ministry_net_prod:
    driver: bridge
